<!--iframe auto-resize (prevents modal/footer being cut on mobile)-->
<script>
(function(){
  const iframe = document.getElementById('edaBomIframe');
  if(!iframe) return;

  const MIN_H = 900;
  const MAX_H = 2400;

  window.addEventListener('message', function(ev){
    try{
      const d = ev.data;
      if(!d || d.type !== 'EDA_BOM_IFRAME_RESIZE') return;
      if(!d.height) return;
      const h = Math.max(MIN_H, Math.min(MAX_H, Number(d.height)));
      iframe.style.height = h + 'px';
    }catch(e){}
  });

  iframe.addEventListener('load', function(){
    iframe.style.height = Math.max(MIN_H, iframe.clientHeight || 0) + 'px';
  });
})();
</script>
